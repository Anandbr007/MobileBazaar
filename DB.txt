TABLE QUERY

-- Create the login table
CREATE TABLE login (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(30),
    password VARCHAR(50),
    role VARCHAR(10)
);

-- Create the customer table
CREATE TABLE customer (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(30),
    email VARCHAR(50),
    phone VARCHAR(30),
    username VARCHAR(30),
    password VARCHAR(50),
    loginid INT,
    FOREIGN KEY (loginid) REFERENCES login(id)
);
